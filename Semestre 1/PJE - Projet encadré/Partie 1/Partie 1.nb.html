Question 3 : Realize the function scoring of parameter N. N corresponds to N-Fold. The function returns the decision trees and the quality of the decision trees and their average.

scoring <- function(n) {
  # on prend une seed afin que nos rsultats soient reproductibles.
  set.seed(1234)
  
  # on instantie la probabilit de nos futurs chantillons d'entranement et de test
  train_prob <- (n - 1) / n
  test_prob  <-      1  / n
  
  # on dtermine ce que l'on veut prdire et en fonction de quelles donnes
  myFormula <- Species ~ Sepal.Length + Sepal.Width + Petal.Length + Petal.Width
  
  # on dfinie le vecteur des qualits de nos connaissances
  quality = c()
  
  for(i in 1:n) {
    # on rpartit nos donnes en un set d'entranement et un set de test
    ind <- sample(2, nrow(iris), replace=TRUE, prob=c(train_prob, test_prob))
    train.data <- iris[ind==1,]
    test.data  <- iris[ind==2,]
    
    # on entrane un arbre de dcision
    # on vous a laiss les dernires valeurs de ctree_controls avec lesquelles on a jou ;)
    iris_ctree <- ctree(formula=myFormula, data=train.data, controls=ctree_control(minsplit=50, minbucket=30, maxsurrogate=3, maxdepth=3))
    
    # on affiche l'arbre
    plot(iris_ctree)
    
    # on prdit les donnes de test
    pred <- predict(iris_ctree, newdata=test.data)
    real <- test.data$Species
    table(pred, real) # ne sera pas affich mais a t utilis
    
    # on save le mean absolute error entre les donnes relles et les donnes prdites
    quality <- c(quality, mae(as.numeric(real), as.numeric(pred)))
  }
  
  return(mean(quality))
}
